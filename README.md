## Install

```bash
npm install --save use-stripe-cart
```

## Usage

```jsx
import React, { Component } from 'react'

import { useMyHook } from 'use-stripe-cart'

const Example = () => {
  const example = useMyHook()
  return (
    <div>{example}</div>
  )
}
```
